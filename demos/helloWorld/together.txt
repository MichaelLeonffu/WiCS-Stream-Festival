//helloWorld.cpp C++

#include <iostream>
using namespace std;

int main(){

	bool goodDay = true;
	if(goodDay)
		cout << "hello world!" << endl;
	else
		cout << "bye world." << endl;

	return 0;
}

//result "hello world!"

}
